package com.formatter
{
/**
 * Performs specific date formatting.
 * 
 * @see formatDuration
 */
public class DateDurationFormatter
{
    
    /**
     * Calculates amount of days in a given year. It can be 365 or 366.
     * 
     * @param year Year e.g. 2011.
     * 
     * @return Amount of days e.g. 365 for 2011 year.
     */
    public static function getDaysInYear(year:int):int
    {
        return isLeapYear(year) ? 366 : 365;
    }
    
    /**
     * Checks if <code>year</code> is a leap-year (has 1 day more) or not.
     * 
     * @return <code>true</code> for leap-years.
     */
    public static function isLeapYear(year:int):Boolean
    {
        if (year % 400 == 0)
            return true;
        else if (year % 100 == 0)
            return false;
        else if (year % 4 == 0)
            return true;
        return false;
    }
    
    /**
     * Calculates amount of days in month.
     * 
     * @param month Month from 0 to 11.
     * 
     * @param year Year e.g. 2011.
     * 
     * @return Amount of days in month e.g. 28 or 31.
     */
    public static function getDaysInMonthInYear(month:int, year:int):int
    {
        month++;
        if (month == 2)
            return getDaysInYear(year) == 365 ? 28 : 29;
        else if (month == 9 || month == 4 || month == 6 || month == 11)
            return 30;
        return 31;
    }
    
    /**
     * Constructor.
     * 
     * @param dateLabelsSource Source of date part labels.
     */
    public function DateDurationFormatter(dateLabelsSource:IDateLabelsSource = null)
    {
        if (!dateLabelsSource)
            dateLabelsSource = new DateLabelsSourceEnglish();
        
        this.dateLabelsSource = dateLabelsSource;
    }
    
    private var dateLabelsSource:IDateLabelsSource;
    
    /**
     * Calculates difference between two dates.
     * 
     * @param initialDate Initial date.
     * 
     * @param finalDate Final date.
     * 
     * @return Duration of time period between two dates. 
     */
    public function dateDifference(initialDate:Date, finalDate:Date):Date
    {
        // make sure than final date is after initial, otherwize swap them
        if (finalDate.time < initialDate.time)
        {
            var tempDate:Date = finalDate;
            finalDate = initialDate;
            initialDate = tempDate;
        }
        
        var years:int = finalDate.getFullYear() - initialDate.getFullYear();
        var months:int = finalDate.getMonth() - initialDate.getMonth();
        var days:int = finalDate.getDate() - initialDate.getDate();
        var hours:int = finalDate.getHours() - initialDate.getHours();
        var minutes:int = finalDate.getMinutes() - initialDate.getMinutes();
        var seconds:int = finalDate.getSeconds() - initialDate.getSeconds();
        var milliseconds:int = finalDate.getMilliseconds() - initialDate.getMilliseconds();
        
        if (milliseconds < 0)
        {
            seconds--;
            milliseconds += 1000;
        }
        if (seconds < 0)
        {
            minutes--;
            seconds += 60;
        }
        if (minutes < 0)
        {
            hours--;
            minutes += 60;
        }
        if (hours < 0)
        {
            days--;
            hours += 24;
        }
        if (days < 0)
        {
            months--;
            
            // calculate amount of days in previous month in finalDate
            var finalMonth:int = finalDate.getMonth() - 1;
            var finalYear:int = finalDate.getFullYear();
            if (finalMonth < 0)
            {
                finalMonth = 11;
                finalYear--;
            }
            days += getDaysInMonthInYear(finalMonth, finalYear);
        }
        if (months < 0)
        {
            years--;
            months += 12;
        }
        
        var date:Date = new Date(0, 0, 1, hours, minutes, seconds, milliseconds);
        date.setFullYear(years, months, days + 1);
        
        return date;
    }
    
    /**
     * Calculates difference between two dates and formats it.
     * 
     * @param initialDate Initial date.
     * 
     * @param finalDate Final date.
     * 
     * @param dateParts Array of <code>String</code> date parts described in
     * <code>DatePart</code> class e.g. <code>[ DatePart.YEAR, DatePart.MONTH ]</code>.
     * If not specified, all non-zero parts are shown.
     * 
     * @return Localized date string.
     */
    public function formatDateDifference(initialDate:Date, finalDate:Date, dateParts:Array = null):String
    {
        return formatDuration(dateDifference(initialDate, finalDate), dateParts);
    }
    
    /**
     * Converts given <code>date</code> duration to localized string e.g. to
     * "34 years 2 months 1 hour 3 milliseconds".
     * 
     * @param date Date to format.
     * 
     * @param dateParts Array of <code>String</code> date parts described in
     * <code>DatePart</code> class e.g. <code>[ DatePart.YEAR, DatePart.MONTH ]</code>.
     * If not specified, all non-zero parts are shown.
     * 
     * @return Localized date string.
     */
    public function formatDuration(duration:Date, dateParts:Array = null):String
    {
        if (!dateParts || dateParts.length == 0)
            dateParts = DatePart.ALL;
        
        var amounts:Object = {};
        amounts[DatePart.YEAR] = duration.getFullYear();
        amounts[DatePart.MONTH] = duration.getMonth();
        amounts[DatePart.DAY] = duration.getDate() - 1;
        amounts[DatePart.HOUR] = duration.getHours();
        amounts[DatePart.MINUTE] = duration.getMinutes();
        amounts[DatePart.SECOND] = duration.getSeconds();
        amounts[DatePart.MILLISECOND] = duration.getMilliseconds();
        var results:Array = [];
        var n:int = dateParts.length;
        for (var i:int = 0; i < n; i++)
        {
            var includeZero:Boolean = i == n - 1 && results.length == 0;
            addDatePartResult(results, amounts[dateParts[i]], dateParts[i], includeZero);
        }
        
        return results.join(" ");
    }
    
    /**
     * @private
     */
    private function addDatePartResult(results:Array, amount:Number, 
                                       datePart:String, includeZero:Boolean = false):void
    {
        // date.getFullYear() returns Number but it's always integer
        amount = int(amount);
        if (amount == 0 && !includeZero)
            return;
        
        results.push(amount.toString() + " " + dateLabelsSource.getLabel(amount, datePart));
    }
    
}
}
