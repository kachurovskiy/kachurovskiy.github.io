package com.formatter
{
/**
 * Languages have different logic of naming different amount of years, hours,
 * minutes and seconds. E.g. in English we write "1 hour" but "2 hours" and "0 hours" 
 * and in Russian the difference is even more dramatic.
 * 
 * <p>Interface for language date labels source.</p>
 */
public interface IDateLabelsSource
{
    
    /**
     * Provides localized date part label.
     * 
     * @param amount Amount of years, months or other date parts.
     * 
     * @param datePart Type of date part e.g. <code>DatePart.YEAR</code>.
     * 
     * @return localized label for year, month or other date part.
     */
    function getLabel(amount:int, datePart:String):String;
    
}
}
