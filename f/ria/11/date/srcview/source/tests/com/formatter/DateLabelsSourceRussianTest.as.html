package com.formatter
{
import flexunit.framework.Assert;

import org.flexunit.asserts.assertEquals;

public class DateLabelsSourceRussianTest
{    
    
    private var source:DateLabelsSourceRussian;
    
    [Before]public function setUp():void
    {
        source = new DateLabelsSourceRussian();
    }
    
    [After]public function tearDown():void
    {
        source = null;
    }
    
    [Test]public function testGetLabel():void
    {
        check("год", -1, DatePart.YEAR);
        check("лет", 0, DatePart.YEAR);
        check("год", 1, DatePart.YEAR);
        check("года", 2, DatePart.YEAR);
        check("года", 3, DatePart.YEAR);
        check("года", 4, DatePart.YEAR);
        var i:int;
        for (i = 5; i <= 20; i++)
        {
            check("лет", i, DatePart.YEAR);
        }
        check("год", 21, DatePart.YEAR);
        check("года", 22, DatePart.YEAR);
        check("года", 23, DatePart.YEAR);
        check("года", 24, DatePart.YEAR);
        for (i = 25; i <= 30; i++)
        {
            check("лет", i, DatePart.YEAR);
        }
    }
    
    private function check(result:String, amount:int, datePart:String):void
    {
        assertEquals(result, source.getLabel(amount, datePart));
    }
}
}
