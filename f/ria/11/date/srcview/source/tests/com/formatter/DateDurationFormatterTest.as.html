package com.formatter 
{
import org.flexunit.asserts.assertEquals;

public class DateDurationFormatterTest
{
    
    private var dateFormatter:DateDurationFormatter;
    
    [Before]public function setUp():void
    {
        dateFormatter = new DateDurationFormatter(new DateLabelsSourceTest());
    }
    
    [After]public function tearDown():void
    {
        dateFormatter = null;
    }
    
    [Test]public function testFormatDateDifference():void
    {
        var nowDate:Date = new Date();
        assertEquals("0 m", dateFormatter.formatDateDifference(nowDate, nowDate));
        
        assertEquals("1 y", dateFormatter.formatDateDifference(
            new Date(2011, 0, 1), new Date(2012, 0, 1)));
        assertEquals("1 m 2 d 3 h 4 m 5 s 6 m", dateFormatter.formatDateDifference(
            new Date(2011, 1, 1, 1, 1, 1, 1), new Date(2011, 2, 3, 4, 5, 6, 7)));
        
        assertEquals("1 y", dateFormatter.formatDateDifference(
            new Date(2012, 0, 1), new Date(2011, 0, 1)));
        
        // 2011 is not leap-year, feb has 28 days
        assertEquals("27 d", dateFormatter.formatDateDifference(
            new Date(2011, 1, 2, 0, 0, 0, 0), new Date(2011, 2, 1, 0, 0, 0, 0)));
        // 2000 is a leap-year, feb has 29 days
        assertEquals("28 d", dateFormatter.formatDateDifference(
            new Date(2000, 1, 2, 0, 0, 0, 0), new Date(2000, 2, 1, 0, 0, 0, 0)));
        
        assertEquals("26 d 22 h 58 m 58 s 999 m", dateFormatter.formatDateDifference(
            new Date(2011, 1, 2, 1, 1, 1, 1), new Date(2011, 2, 1, 0, 0, 0, 0)));
        
        // 2011 is not leap-year, feb has 28 days
        assertEquals("1 y 23 d", dateFormatter.formatDateDifference(
            new Date(2010, 1, 10, 0, 0, 0, 0), new Date(2011, 2, 5, 0, 0, 0, 0)));
    }
    
    [Test]public function testFormatDuration():void
    {
        var devDate:Date = new Date(2011, 3, 12, 19, 23, 1, 1);
        
        testOneFormatDuration("2011 y 3 m 11 d 19 h 23 m 1 s 1 m", devDate);
        testOneFormatDuration("2011 y 3 m 11 d 19 h 23 m 1 s 1 m", devDate, []);
        
        testOneFormatDuration("2011 y", devDate, [ DatePart.YEAR ]);
        testOneFormatDuration("2011 y 3 m", devDate, 
            [ DatePart.YEAR, DatePart.MONTH ]);
        testOneFormatDuration("2011 y 3 m 11 d", devDate, 
            [ DatePart.YEAR, DatePart.MONTH, DatePart.DAY ]);
        testOneFormatDuration("2011 y 3 m 11 d 19 h", devDate, 
            [ DatePart.YEAR, DatePart.MONTH, DatePart.DAY, DatePart.HOUR ]);
        testOneFormatDuration("2011 y 3 m 11 d 19 h 23 m", devDate, 
            [ DatePart.YEAR, DatePart.MONTH, DatePart.DAY, DatePart.HOUR, DatePart.MINUTE ]);
        testOneFormatDuration("2011 y 3 m 11 d 19 h 23 m 1 s", devDate, 
            [ DatePart.YEAR, DatePart.MONTH, DatePart.DAY, DatePart.HOUR, DatePart.MINUTE, DatePart.SECOND ]);
        testOneFormatDuration("2011 y 3 m 11 d 19 h 23 m 1 s 1 m", devDate, 
            [ DatePart.YEAR, DatePart.MONTH, DatePart.DAY, DatePart.HOUR, DatePart.MINUTE, DatePart.SECOND, DatePart.MILLISECOND ]);
        
        testOneFormatDuration("3 m 2011 y", devDate,
            [ DatePart.MONTH, DatePart.YEAR ]);
        
        var zeroDate:Date = new Date(0, 0, 1, 0, 0, 0, 0);
        zeroDate.setFullYear(0, 0, 1);
        testOneFormatDuration("0 m", zeroDate);
        testOneFormatDuration("0 y", zeroDate, [ DatePart.MONTH, DatePart.YEAR ]);
    }
    
    private function testOneFormatDuration(formattedDate:String, date:Date, parts:Array = null):void
    {
        assertEquals(formattedDate, dateFormatter.formatDuration(date, parts));
    }
    
    [Test]public function testIsLeapYear():void
    {
        assertEquals(true, DateDurationFormatter.isLeapYear(2000));
        assertEquals(false, DateDurationFormatter.isLeapYear(2001));
        assertEquals(false, DateDurationFormatter.isLeapYear(2002));
        assertEquals(false, DateDurationFormatter.isLeapYear(2003));
        assertEquals(true, DateDurationFormatter.isLeapYear(2004));
        assertEquals(false, DateDurationFormatter.isLeapYear(2100));
        assertEquals(false, DateDurationFormatter.isLeapYear(2200));
        assertEquals(false, DateDurationFormatter.isLeapYear(2300));
        assertEquals(true, DateDurationFormatter.isLeapYear(2400));
    }
    
    [Test]public function testGetDaysInYear():void
    {
        assertEquals(366, DateDurationFormatter.getDaysInYear(2000));
        assertEquals(365, DateDurationFormatter.getDaysInYear(2001));
    }
    
    [Test]public function testGetDaysInMonthInYear():void
    {
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(0, 2011));
        assertEquals(28, DateDurationFormatter.getDaysInMonthInYear(1, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(2, 2011));
        assertEquals(30, DateDurationFormatter.getDaysInMonthInYear(3, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(4, 2011));
        assertEquals(30, DateDurationFormatter.getDaysInMonthInYear(5, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(6, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(7, 2011));
        assertEquals(30, DateDurationFormatter.getDaysInMonthInYear(8, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(9, 2011));
        assertEquals(30, DateDurationFormatter.getDaysInMonthInYear(10, 2011));
        assertEquals(31, DateDurationFormatter.getDaysInMonthInYear(11, 2011));
        
        // february in leap-year has 29 days
        assertEquals(29, DateDurationFormatter.getDaysInMonthInYear(1, 2012));
    }
    
}
}

import com.formatter.IDateLabelsSource;

/**
 * Simple date part labels source for tests only.
 */
class DateLabelsSourceTest implements IDateLabelsSource
{
    
    public function getLabel(amount:int, type:String):String
    {
        // return "y" for "year", "d" for "day" and so on
        return type.charAt(0);
    }
    
}
