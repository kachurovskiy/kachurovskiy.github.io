package com.formatter
{
import flexunit.framework.Assert;

import org.flexunit.asserts.assertEquals;

public class DateLabelsSourceEnglishTest
{    
    
    private var source:DateLabelsSourceEnglish;
    
    [Before]public function setUp():void
    {
        source = new DateLabelsSourceEnglish();
    }
    
    [After]public function tearDown():void
    {
        source = null;
    }
    
    [Test]public function testGetLabel():void
    {
        var i:int;
        for (i = -20; i < 20; i++)
        {
            var suffix:String = Math.abs(i) != 1 ? "s" : "";
            check("year" + suffix, i, DatePart.YEAR);
            check("month" + suffix, i, DatePart.MONTH);
            check("day" + suffix, i, DatePart.DAY);
            check("hour" + suffix, i, DatePart.HOUR);
            check("minute" + suffix, i, DatePart.MINUTE);
            check("second" + suffix, i, DatePart.SECOND);
            check("millisecond" + suffix, i, DatePart.MILLISECOND);
            check("?" + suffix, i, "wrong string");
        }
    }
    
    private function check(result:String, amount:int, datePart:String):void
    {
        assertEquals(result, source.getLabel(amount, datePart));
    }
}
}
