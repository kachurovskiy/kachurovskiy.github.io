<?xml version="1.0" encoding="utf-8"?>
<s:ItemRenderer xmlns:fx="http://ns.adobe.com/mxml/2009" 
    xmlns:s="library://ns.adobe.com/flex/spark" 
    xmlns:mx="library://ns.adobe.com/flex/mx" 
    autoDrawBackground="true" width="50" height="50"
    toolTip="{getToolTip(ImageData(data).color)}">
    
    <fx:Script>
    <![CDATA[
        import sample.model.Color;
        import sample.model.ImageData;
        
        public function getToolTip(color:Color):String
        {
            if (!color)
                return null;
            return "Hue: " + color.hue + "\nSaturation: " + color.saturation + 
                "\nBrightness: " + color.brightness;
        }
        
        
    ]]>
    </fx:Script>
    
    <s:Rect left="0" right="0" top="0" bottom="0" visible="{ImageData(data).color != null}">
        <s:fill>
            <s:SolidColor color="{ImageData(data).color.toUInt()}"/>
        </s:fill>
    </s:Rect>
    
    <s:BitmapImage top="2" bottom="2" left="2" right="2" source="{ImageData(data).thumbnail}" 
        scaleMode="letterbox"/>
    
</s:ItemRenderer>
