package sample.model
{
import be.boulevart.google.ajaxapi.search.images.data.GoogleImage;

import flash.display.BitmapData;
import flash.events.EventDispatcher;

/**
 * Contains data about the image provided by Google and loaded thumbnail BitmapData.
 */
public class ImageData extends EventDispatcher
{
    
    public function ImageData(googleImage:GoogleImage)
    {
        super();
        
        this.googleImage = googleImage;
    }
    
    [Bindable]public var googleImage:GoogleImage;
    
    [Bindable]
    /**
     * Is true while thumbnail is being loaded.
     */
    public var loadingThumbnail:Boolean = false;
    
    [Bindable]public var thumbnail:BitmapData;
    
    [Bindable]
    /**
     * Prevailing color of the thumbnail. Calculated on the client.
     */
    public var color:Color;
    
    [Bindable]
    /**
     * Value of color.hue.
     */
    public var colorHue:Number;
    
}
}
