package sample.actions
{
import flash.display.Bitmap;
import flash.display.BitmapData;
import flash.display.Loader;
import flash.events.ErrorEvent;
import flash.events.Event;
import flash.events.IOErrorEvent;
import flash.events.SecurityErrorEvent;
import flash.net.URLLoader;
import flash.net.URLLoaderDataFormat;
import flash.net.URLRequest;
import flash.utils.ByteArray;

import sample.clusterization.kMeans.KMeansCluster;
import sample.clusterization.kMeans.KMeansClusterizationMethod;
import sample.model.Color;
import sample.model.ImageData;

/**
 * Fills <code>ImageData</code>-s <code>thumbnail</code> property with <code>BitmapData</code>.
 * Also fills <code>ImageData.color</code>.
 */
public class LoadThumbnailAction
{
    
    private var urlLoader:URLLoader;
    
    private var loader:Loader;
    
    private var imageData:ImageData;
    
    public function start(imageData:ImageData):void
    {
        this.imageData = imageData;
        
        urlLoader = new URLLoader();
        urlLoader.dataFormat = URLLoaderDataFormat.BINARY;
        urlLoader.addEventListener(Event.COMPLETE, urlLoader_completeHandler);
        urlLoader.addEventListener(IOErrorEvent.IO_ERROR, urlLoader_errorHandler);
        urlLoader.addEventListener(SecurityErrorEvent.SECURITY_ERROR, urlLoader_errorHandler);
        urlLoader.load(new URLRequest(imageData.googleImage.thumbUrl));
    }
    
    private function urlLoader_completeHandler(event:Event):void
    {
        loadBytes(urlLoader.data);
    }
    
    private function urlLoader_errorHandler(event:ErrorEvent):void
    {
        trace(event.text);
    }
    
    private function loadBytes(byteArray:ByteArray):void
    {
        loader = new Loader();
        loader.contentLoaderInfo.addEventListener(Event.INIT, loader_initHandler);
        loader.contentLoaderInfo.addEventListener(IOErrorEvent.IO_ERROR, loader_errorHandler);
        loader.loadBytes(byteArray);
    }
    
    private function loader_initHandler(event:Event):void
    {
        imageData.loadingThumbnail = false;
        try
        {
            var bitmapData:BitmapData = Bitmap(loader.content).bitmapData;
            imageData.thumbnail = bitmapData;
        }
        catch (error:Error)
        {
            trace(error.message);
        }
        
        // calculate color
        if (imageData.thumbnail)
        {
            var method:KMeansClusterizationMethod = new KMeansClusterizationMethod();
            method.addEventListener(Event.COMPLETE, method_completeHandler);
            method.startFromBitmapData(imageData.thumbnail);
        }
    }
    
    private function loader_errorHandler(event:ErrorEvent):void
    {
        imageData.loadingThumbnail = false;
        trace(event.text);
    }
    
    private function method_completeHandler(event:Event):void
    {
        var method:KMeansClusterizationMethod = KMeansClusterizationMethod(event.target);
        // select someting not black and not white if possible
        var black:Color = Color.fromUInt(0x000000);
        var white:Color = Color.fromUInt(0xFFFFFF);
        var minDistance:Number = Color.distance(black, white) / 5;
        var selectedColor:Color;
        var selectedColorClusterSize:int = 0;
        for each (var cluster:KMeansCluster in method.clusters)
        {
            var color:Color = cluster.center;
            if (Color.distance(black, color) > minDistance &&
                Color.distance(white, color) > minDistance &&
                cluster.size > selectedColorClusterSize)
            {
                selectedColor = color;
                selectedColorClusterSize = cluster.size;
                break;
            }
        }
        if (!selectedColor)
            selectedColor = method.clusters[0].center;
        
        imageData.color = selectedColor;
        imageData.colorHue = selectedColor.hue;
    }
    
    
}
}
