<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" 
    xmlns:local="*" horizontalScrollPolicy="off" viewSourceURL="srcview/index.html">

    <mx:VBox paddingBottom="10" paddingLeft="10" paddingRight="10" 
        paddingTop="10" horizontalGap="10" width="100%" height="100%">
        
        <mx:Panel title="The simplest empty renderer">
        
            <local:TestingComponent id="component1" itemRenderer="{EmptyItemRenderer}"
                started="startedHandler(event);"/>
                
        </mx:Panel>
        
        <mx:Panel title="The simplest custom item renderer">
        
            <local:TestingComponent id="component2" itemRenderer="{CustomSimpleItemRenderer}"
                started="startedHandler(event);"/>
                
        </mx:Panel>
        
        <mx:Panel title="Standart Adobe renderer">
        
            <local:TestingComponent id="component3" started="startedHandler(event);"/>
            
        </mx:Panel>
        
        <mx:Panel title="Custom easy MXML renderer">
        
            <local:TestingComponent id="component4" itemRenderer="{EasyRenderer}"
                started="startedHandler(event);"/>
                
        </mx:Panel>
        
    </mx:VBox>
    
<mx:Script>
<![CDATA[
    import renderers.CustomSimpleItemRenderer;

    import renderers.EmptyItemRenderer;
    import renderers.EasyRenderer;
    
    private var selectedComponent:TestingComponent;
    
    private function startedHandler(event:TestingComponentEvent):void
    {
        selectedComponent = TestingComponent(event.target);
        
        var components:Array = [component1, component2, component3, component4];
        for each (var component:TestingComponent in components)
        {
            if (component != selectedComponent)
                component.minimized = true;
        }
    }
    
]]>
</mx:Script>

</mx:Application>
