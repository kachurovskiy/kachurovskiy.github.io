<?xml version="1.0" encoding="utf-8"?>
<mx:VBox xmlns:mx="http://www.adobe.com/2006/mxml" width="420"
    horizontalAlign="center" paddingBottom="10" paddingLeft="10" paddingRight="10" 
    paddingTop="10">
    
<mx:Metadata>
    
    [Event(name="started", type="TestingComponentEvent")]
    
</mx:Metadata>
    
    <mx:Button label="Start test{ averageTime == 0 ? '' : ' (avg ' + averageTime + ')' }" 
        click="startTest();" enabled="{endTime > startTime || startTime == 0}"/>
    
    <mx:states>
        
        <mx:State name="full">
            
            <mx:AddChild>
                
                <mx:DataGrid id="dataGrid" width="100%" visible="{!minimized}" 
                    includeInLayout="{!minimized}"/>
                
            </mx:AddChild>
            
            <mx:AddChild>
                
                <mx:Label id="currentTimeLabel" fontSize="20"
                    text="Time for initialization: {endTime - startTime}"/>
                
            </mx:AddChild>
            
            <mx:AddChild>
                
                <mx:Label id="averageTimeLabel" text="Average: {averageTime}" 
                    fontSize="14"/>
                
            </mx:AddChild>
            
        </mx:State>
        
    </mx:states>
        
<mx:Script>
<![CDATA[

    import mx.events.FlexEvent;

    //--------------------------------------------------------------------------
    //
    //  Properties
    //
    //--------------------------------------------------------------------------
    
    public var itemRenderer:Class;
    
    [Bindable]
    public var averageTime:Number = 0;

    //----------------------------------
    //  minimized
    //----------------------------------
    
    private var _minimized:Boolean = true;
    
    [Bindable]
    /**
     * Ability to minimize component if it's not in use.
     */
    public function get minimized():Boolean
    {
        return _minimized;
    }
    
    public function set minimized(value:Boolean):void
    {
        _minimized = value;
        
        currentState = _minimized ? "" : "full";
    }
    
    [Bindable]
    private var startTime:Number = 0;
    [Bindable]
    private var endTime:Number = 0;
    
    /**
     * Here we store all time results to calculate average time.
     */
    private var times:Array = [];

    private var firstTime:Boolean = true;

    //--------------------------------------------------------------------------
    //
    //  Methods
    //
    //--------------------------------------------------------------------------
    
    private function startTest():void
    {
        dispatchEvent(new TestingComponentEvent(TestingComponentEvent.STARTED));
        
        minimized = false;
        
        var array:Array = [];
        var textPart:String = "Some ";
        for (var i:int = 0; i < 1000; i++)
        {
            var object:Object =
                {
                    text1: textPart + i++,
                    text2: textPart + i++,
                    text3: textPart + i++,
                    text4: textPart + i
                }
            array.push(object);
        }
        
        if (itemRenderer)
            dataGrid.itemRenderer = new ClassFactory(itemRenderer);
        
        dataGrid.dataProvider = array;
        dataGrid.addEventListener(FlexEvent.UPDATE_COMPLETE, 
            dataGrid_updateCompleteHandler);

        startTime = (new Date()).time;
    }

    //--------------------------------------------------------------------------
    //
    //  Event handlers
    //
    //--------------------------------------------------------------------------

    private function dataGrid_updateCompleteHandler(event:FlexEvent):void
    {
        dataGrid.removeEventListener(FlexEvent.UPDATE_COMPLETE, 
            dataGrid_updateCompleteHandler);
        
        endTime = (new Date()).time;
        
        if (firstTime)
        {
            firstTime = false;
            startTest();
            return;
        }
        
        times.push(endTime - startTime);
        var summ:Number = 0;
        for each (var number:Number in times)
            summ += number;
        averageTime = Math.round(summ / times.length);
    }
    
]]>
</mx:Script>
</mx:VBox>
