<?xml version="1.0" encoding="utf-8"?>
<mx:HBox xmlns:mx="http://www.adobe.com/2006/mxml" initialize="init();"
    addedToStage="if (initialized) init();">
    
<mx:Script>
<![CDATA[
    import mx.core.BitmapAsset;
    
    public static function getRandomBitmapData():BitmapData
    {
        // create transparent bitmap
        var bitmapData:BitmapData = new BitmapData(16, 16, true, 0);
        var seed:Number = Math.floor(Math.random() * 10);
        var channels:uint = BitmapDataChannel.RED | BitmapDataChannel.BLUE;
        bitmapData.perlinNoise(100 * Math.random(), 100 * Math.random(), 10 * Math.random(), 
            seed, false, true, channels, false, null);
        return bitmapData;
    }
    
    public function get selectedBitmapData():BitmapData
    {
        return image.source.bitmapData;
    }
    
    private function init():void
    {
        image.source = new BitmapAsset(getRandomBitmapData());
    }
    
]]>
</mx:Script>
    
    <mx:Image id="image"/>
    
</mx:HBox>
