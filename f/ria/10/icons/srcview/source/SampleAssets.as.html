package
{
/**
 * This class shows how your application can easily work with assets that resize
 * in external ZIP archive.
 * 
 * <p>It is possible to seamlessly create icon from layers, e.g. see
 * how <code>addBarChart48</code> is created from barChart48 and addLayer48.</p>
 */
public class SampleAssets extends AssetsBase
{
    
    public function SampleAssets(assetManager:AssetManager)
    {
        super(assetManager);
    }
    
    //--------------------------------------------------------------------------
    //
    //  Just icons from assets.zip
    //
    //--------------------------------------------------------------------------
    
    [Bindable("barChart48Change")]
    public var barChart48:Class;

    [Bindable("pieChart48Change")]
    public var pieChart48:Class;
    
    [Bindable("comment48Change")]
    public var comment48:Class;
    
    [Bindable("addLayer48Change")]
    public var addLayer48:Class;
    
    [Bindable("infoLayer48Change")]
    public var infoLayer48:Class;
    
    [Bindable("deleteLayer48Change")]
    public var deleteLayer48:Class;
    
    [Bindable("addLayerAnimated48Change")]
    public var addLayerAnimated48:Class;
    
    //--------------------------------------------------------------------------
    //
    //  Layered icons
    //
    //--------------------------------------------------------------------------
    
    [IconLayer("barChart48")]
    [IconLayer("addLayer48")]
    [Bindable("addBarChart48Change")]
    public var addBarChart48:Class;
    
    [IconLayer("barChart48")]
    [IconLayer("infoLayer48")]
    [Bindable("infoBarChart48Change")]
    public var infoBarChart48:Class;
    
    [IconLayer("barChart48")]
    [IconLayer("deleteLayer48")]
    [Bindable("deleteBarChart48Change")]
    public var deleteBarChart48:Class;
    
    [IconLayer("pieChart48")]
    [IconLayer("addLayer48")]
    [Bindable("addPieChart48Change")]
    public var addPieChart48:Class;
    
    [IconLayer("pieChart48")]
    [IconLayer("infoLayer48")]
    [Bindable("infoPieChart48Change")]
    public var infoPieChart48:Class;
    
    [IconLayer("pieChart48")]
    [IconLayer("deleteLayer48")]
    [Bindable("deletePieChart48Change")]
    public var deletePieChart48:Class;
    
    [IconLayer("comment48")]
    [IconLayer("addLayer48")]
    [Bindable("addComment48Change")]
    public var addComment48:Class;
    
    [IconLayer("comment48")]
    [IconLayer("infoLayer48")]
    [Bindable("infoComment48Change")]
    public var infoComment48:Class;
    
    [IconLayer("comment48")]
    [IconLayer("deleteLayer48")]
    [Bindable("deleteComment48Change")]
    public var deleteComment48:Class;
    
    [IconLayer("comment48")]
    [IconLayer("addLayerAnimated48")]
    [Bindable("addComment48Change")]
    public var addCommentAnimated48:Class;
    
}
}
