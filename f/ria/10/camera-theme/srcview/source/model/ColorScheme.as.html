package model
{
import mx.utils.ColorUtil;

/**
 *  Color scheme for the Flex 4 application.
 */
public class ColorScheme
{
    
    //--------------------------------------------------------------------------
    //
    //  Static properties
    //
    //--------------------------------------------------------------------------

    public static const COLORS_NEEDED:int = 8;
    
    //--------------------------------------------------------------------------
    //
    //  Static methods
    //
    //--------------------------------------------------------------------------

    public static function createSchemesFromColors(colors:Vector.<Color>):Vector.<ColorScheme>
    {
        var result:Vector.<ColorScheme> = new Vector.<ColorScheme>();
        
        var darkSortedColors:Vector.<Color> = colors.sort(darkCompareFunction);
        var colorScheme:ColorScheme = new ColorScheme(darkSortedColors);
        
        var color:Color = darkSortedColors.shift();
        if (color)
        {
            colorScheme.color = color.toUInt();
            colorScheme.symbolColor = ColorUtil.adjustBrightness(colorScheme.color, 15);
        }
        
        color = darkSortedColors.shift();
        if (color)
            colorScheme.accentColor = color.toUInt();
        
        color = darkSortedColors.shift();
        if (color)
            colorScheme.selectionColor = color.toUInt();
        
        color = darkSortedColors.shift();
        if (color)
            colorScheme.rollOverColor = ColorUtil.adjustBrightness(color.toUInt(), 10);

        color = darkSortedColors.shift();
        if (color)
            colorScheme.textRollOverColor = color.toUInt();
            
        color = darkSortedColors.shift();
        if (color)
            colorScheme.contentBackgroundColor = color.toUInt();
        
        color = darkSortedColors.shift();
        if (color)
            colorScheme.focusColor = color.toUInt();
        
        color = darkSortedColors.shift();
        if (color)
            colorScheme.chromeColor = color.toUInt();
        
        result.push(colorScheme);
        
        return result;
    }
    
    private static function darkCompareFunction(color1:Color, color2:Color):int
    {
        var summ1:int = color1.r + color1.g + color1.b;
        var summ2:int = color2.r + color2.g + color2.b;
        if (summ1 < summ2)
            return -1;
        else if (summ1 == summ2)
            return 0;
        else
            return 1;
    }
    
    public function ColorScheme(colors:Vector.<Color>)
    {
        sourceColors = new Vector.<uint>();
        var n:int = colors.length;
        for (var i:int = 0; i < n; i++)
        {
            sourceColors.push(colors[i].toUInt());
        }
    }
    
    public var sourceColors:Vector.<uint>;
    
    [Bindable]public var chromeColor:uint;
    
    [Bindable]public var focusColor:uint;
    
    [Bindable]public var contentBackgroundColor:uint;
    
    [Bindable]public var symbolColor:uint;
    
    [Bindable]public var accentColor:uint;
    
    [Bindable]public var selectionColor:uint;
    
    [Bindable]public var rollOverColor:uint;
    
    [Bindable]public var textRollOverColor:uint;
    
    [Bindable]public var color:uint;
    
}
}
