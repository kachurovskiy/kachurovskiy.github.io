package model
{

public class Color
{
    
    //--------------------------------------------------------------------------
    //
    //  Static methods
    //
    //--------------------------------------------------------------------------

    /**
     *  @return Distance between two colors.
     */
    public static function distance(color1:Color, color2:Color):Number
    {
        return Math.sqrt(Math.pow(color1.r - color2.r, 2)) +
            Math.sqrt(Math.pow(color1.g - color2.g, 2)) +
            Math.sqrt(Math.pow(color1.b - color2.b, 2));
    }
    
    /**
     *  @return Simple center of a set of colors.
     */
    public static function average(colors:Vector.<Color>):Color
    {
        var r:Number = 0;
        var g:Number = 0;
        var b:Number = 0;
        
        var n:int = colors.length;
        for (var i:int = 0; i < n; i++)
        {
            var color:Color = colors[i];
            r += color.r;
            g += color.g;
            b += color.b;
        }
        r /= n;
        g /= n;
        b /= n;
        return new Color(uint(r), uint(g), uint(b));
    }
    
    public static function fromUInt(color:uint):Color
    {
        var b:uint = color % 256;
        color -= b;
        color /= 256;
        var g:uint = color % 256;
        color -= g;
        color /= 256;
        var r:uint = color % 256;
        return new Color(r, g, b);
    }
    
    //--------------------------------------------------------------------------
    //
    //  Constructor
    //
    //--------------------------------------------------------------------------
    
    public function Color(r:uint, g:uint, b:uint)
    {
        _r = r;
        _g = g;
        _b = b;
        calculateHSV();
    }
    
    //--------------------------------------------------------------------------
    //
    //  Properties
    //
    //--------------------------------------------------------------------------

    private var _r:uint;
    
    public function get r():uint
    {
        return _r;
    }
    
    private var _g:uint;
    
    public function get g():uint
    {
        return _g;
    }
    
    private var _b:uint;
    
    public function get b():uint
    {
        return _b;
    }
    
    public var hue:Number;
    public var saturation:Number;
    public var brightness:Number;
    
    //--------------------------------------------------------------------------
    //
    //  Methods
    //
    //--------------------------------------------------------------------------

    public function toUInt():uint
    {
        return r * 256 * 256 + g * 256 + b;
    }
    
    public function average():Number
    {
        return (_r + _g + _b) / 3;
    }
    
    public function calculateHSV():void
    {
        var r:Number = _r / 255;
        var g:Number = _g / 255;
        var b:Number = _b / 255;
        
        var x:Number = Math.min(r, g, b);
        var v:Number = Math.max(r, g, b);
        if (v == x) 
        {
            hue = 0;
            saturation = 0;
            brightness = v;
            return;
        }

        var f:Number = (r == x) ? (g - b) : ((g == x) ? b - r : r - g);
        var i:Number = (r == x) ? 3 : ((g == x) ? 5 : 1);
        hue = i - f /(v - x);
        saturation = (v - x) / v;
        brightness = v; 
    }
}
}
