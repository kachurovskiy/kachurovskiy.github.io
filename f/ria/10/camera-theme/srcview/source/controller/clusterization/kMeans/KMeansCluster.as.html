package controller.clusterization.kMeans
{
import model.Color;

public class KMeansCluster
{
    
    public function KMeansCluster(center:Color)
    {
        colors.push(center);
        
        this.center = center;
    }    
    
    public var colors:Vector.<Color> = new Vector.<Color>();
    
    public var center:Color;
    
    /**
     *  @return true if center has changed.
     */
    public function evaluateCenter():Boolean
    {
        if (colors.length < 2)
            return false;
        
        var newCenter:Color = Color.average(colors);
        if (newCenter.r != center.r || newCenter.g != center.g || newCenter.b != center.b)
        {
            center = newCenter;
            return true;
        }
        return false;
    }
    
}
}
