package controller.saturation
{
import controller.IExtractMethod;

import flash.events.Event;
import flash.events.EventDispatcher;

import model.Color;
import model.ColorScheme;

[Event("complete", type="flash.events.Event")]
public class SaturationMethod extends EventDispatcher implements IExtractMethod
{
    
    private var _result:Vector.<Color>;
    
    public function get result():Vector.<Color>
    {
        return _result;
    }
    
    public function start(colors:Vector.<Color>):void
    {
        colors = colors.sort(saturationSortFunction);
        
        _result = new Vector.<Color>();
        var n:int = colors.length;
        for (var i:int = 0; i < n; i++)
        {
            var color:Color = colors[i];
            var near:Boolean = false;
            var m:int = _result.length;
            for (var j:int = 0; j < m; j++)
            {
                var distance:Number = Color.distance(_result[j], color);
                if (distance < 60)
                {
                    near = true;
                    break;
                }
            }
            if (!near)
                _result.push(color);
        }
        
        var step:int = Math.floor(_result.length / ColorScheme.COLORS_NEEDED);
        if (step == 0)
            step = 1;
        
        var newResult:Vector.<Color> = new Vector.<Color>();
        for (var k:int = 0; k < _result.length; k += step)
        {
            newResult.push(_result[k]);
        }
        _result = newResult;
        
        dispatchEvent(new Event(Event.COMPLETE));
    }
    
    private static function saturationSortFunction(color1:Color, color2:Color):int
    {
        var value1:Number = color1.saturation * 5 + color1.brightness;
        var value2:Number = color2.saturation * 5 + color2.brightness;
        if (value1 < value2)
            return 1;
        else if (value1 == value2)
            return 0;
        else
            return -1;
    }
    
}
}
