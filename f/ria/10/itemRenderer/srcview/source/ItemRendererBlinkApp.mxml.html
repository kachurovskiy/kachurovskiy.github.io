<?xml version="1.0" encoding="utf-8"?>
<!--
    This application shows why custom Spark item renderers should override
    getCurrentRendererState() if they use custom states.
-->
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
    xmlns:s="library://ns.adobe.com/flex/spark" 
    xmlns:mx="library://ns.adobe.com/flex/mx" minWidth="955" minHeight="600"
    applicationComplete="clearLog();" viewSourceURL="srcview/index.html">
    
    <fx:Script>
    <![CDATA[
        import mx.collections.ArrayCollection;
        
        [Bindable("__NoChangeEvent__")]
        private var dataProvider:ArrayCollection = new ArrayCollection(
            [
                new Data1("Orange"),
                new Data1("Apple"),
                new Data2(15),
                new Data2(42),
                new Data1("Potato")
            ]);
        
        private function clearLog():void
        {
            textArea.text = "";
        }
        
    ]]>
    </fx:Script>
    
    <s:layout>
        <s:HorizontalLayout horizontalAlign="center" verticalAlign="middle"/>
    </s:layout>
    
    <s:VGroup>
        
        <s:Label text="Good"/>
        
        <s:List itemRenderer="GoodItemRenderer" dataProvider="{dataProvider}"/>
        
    </s:VGroup>
    
    <s:VGroup>
        
        <s:Label text="Bad"/>
        
        <s:List itemRenderer="BadItemRenderer" dataProvider="{dataProvider}"/>
    
    </s:VGroup>
    
    <mx:Spacer width="10"/>
    
    <s:VGroup>
        
        <s:Label text="Log:"/>
        
        <s:TextArea id="textArea" width="350" height="120"/>
        
        <s:Button label="Clear log" click="clearLog();"/>
        
    </s:VGroup>
    
</s:Application>
