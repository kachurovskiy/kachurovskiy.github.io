<?xml version="1.0" encoding="utf-8"?>
<s:ItemRenderer xmlns:fx="http://ns.adobe.com/mxml/2009" 
    xmlns:s="library://ns.adobe.com/flex/spark" 
    xmlns:mx="library://ns.adobe.com/flex/mx" 
    autoDrawBackground="true" height="30">
    
    <fx:Script>
    <![CDATA[
        import mx.core.FlexGlobals;
        
        [Bindable("dataChange")]
        private var data1:Data1;
        
        [Bindable("dataChange")]
        private var data2:Data2;
        
        override public function set data(value:Object):void
        {
            data1 = value as Data1;
            data2 = value as Data2;
            
            super.data = value;
            
            currentState = getState();
        }
        
        private function getState():String
        {
            if (data1)
                return "data1";
            else if (data2)
                return "data2";
            return null;
        }
        
        private function label1_removedFromStageHandler(event:Event):void
        {
            if (data1)
                FlexGlobals.topLevelApplication.textArea.appendText("Label " + data1.label + " removed from stage\n");
        }

        private function label1_addedToStageHandler(event:Event):void
        {
            if (data1)
                FlexGlobals.topLevelApplication.textArea.appendText("Label " + data1.label + " added to stage\n");
        }

    ]]>
    </fx:Script>
    
    <s:states>
        <s:State name="common"/><!-- it blinks to the first state declared -->
        <s:State name="data1"/>
        <s:State name="data2"/>
    </s:states>
    
    <s:Label text="{data1.label}" includeIn="data1" verticalCenter="0" left="5"
        removedFromStage="label1_removedFromStageHandler(event)"
        addedToStage="label1_addedToStageHandler(event)"/>
    
    <s:NumericStepper id="stepper" value="{data2.amount}" maximum="100" includeIn="data2"
        change="data2.amount = stepper.value" verticalCenter="0" left="5"/>
    
</s:ItemRenderer>
